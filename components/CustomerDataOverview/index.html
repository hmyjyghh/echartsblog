使用dataModel，dataTotalList，dataDetailList来定义 CustomerDataOverview 的样式

<template>
  <div class="app">
    <CustomerDataOverview  v-if="foo.dataTotalList" :foos="foo"/>
  </div>
</template>
<script lang="ts">

// import { defineComponent } from 'vue'
import { defineComponent, onMounted, reactive, ref } from 'vue'
export default defineComponent({
  name: 'App',
  setup() {
    const foo = ref({
      dataModel: ['流向上海数据概览', '2021-02-28'],
      dataTotalList: [21438600, 28096500],
      dataDetailList: [38093100, 38310900, 26313400, 8.96, 6.99, 9.87]
      })

   
    return {
      foo
    }
  },
})
</script>
