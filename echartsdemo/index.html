1.默认数值展示 #

代码展示

  • 配置series中的label属性
  • label中的show设置为true, 则显示数值
  • formatter函数自定义数值字体大小及是否加粗
label: {
  show: true,
  fontSize: 10,
  color: "#E6262C",
  formatter(v) {
    if (props.unit == "%") {
      if (v.dataIndex === yAxisData.length - 1) {
        return "{a|" + `${v.value}${props.unit}` + "}";
      } else {
        return "{b|" + `${v.value}${props.unit}` + "}";
      }
    } else {
      if (v.dataIndex === yAxisData.length - 1) {
        return "{a|" + v.value + "}";
      } else {
        return "{b|" + v.value + "}";
      }
    }
  },
  rich: { // 定义模板中的字体样式
    a: {
      fontSize: 12,
      fontWeight: "bold",
    },
    b: {
      fontSize: 10,
    },
  },
},

2.近一个月不展示默认数值 #

代码展示

  • 配置series中的label属性
  • label中的show设置为false, 则不显示数值
label: {
  show: false
},
  • 平均线, 虚线展示, 最后一个点展示数值且字号放大，圆点加粗
series: [
  {},
  // 平均线配置
  {
    symbol: (value, ind) => {
      let index = yAxisData2.length - 1;
      if (index == ind.dataIndex) { // 最后一个点，才展示圆圈
        return "circle";
      } else {
        return "none";
      }
    },
    symbolSize: 8, // 圆圈大小
    label: {
      show: true,
      fontSize: 14, // 最后一个点文字大小，加粗设置
      color: "#426DFF",
      position: "bottom",
      padding: [0, 30, 0, 0],
      formatter(v) {
        let index = yAxisData2.length - 1;
        if (index == v.dataIndex) { // 最后一个点展示数值
          return v.value;
        } else {
          return "";
        }
      },
    },
    lineStyle: {
      width: 2,
      type: "dashed", //'dotted'虚线 'solid'实线
    },
  }
]

3.间隔一个点展示数值 #

代码展示

  • 配置series中的label属性
  • label中的show设置为true, 则显示数值
  • formatter函数自定义间隔显示数值
  • rich属性定义模板中的字体大小及加粗效果
series: [
  {
    label: {
      show: true,
      fontSize: 10,
      color: "#E60012",
      position: "top",
      formatter(v) {
        if (trendIndex.value == 2) { // 近15天 
          if (v.dataIndex % 2 == 0) { // 间隔显示数值
            if (v.dataIndex === obj?.agentcount.length - 1) {
              return '{a|' + v.value + '}';
            } else {
              return '{b|' + v.value + '}';
            }
          } else {
            return "";
          }
        }
      },
      rich: {
        a: {
          fontSize: 12,
          fontWeight: 'bold'
        },
        b: {
          fontSize: 10
        }
      },
    },
    symbolSize: (rawValue, params) => {
      params.symbolSize = obj?.sizeArr[params.dataIndex]
      return params.symbolSize
    },
  }
]
